//kernel函数
__global__ void add_3( int *a, int *b, int *c ) {
	int tid = blockIdx.x * blockDim.x + threadIdx.x;
 	while (tid < N) {
 		c[tid] = a[tid] + b[tid];
 		tid += blockDim.x * gridDim.x;
	}
}
